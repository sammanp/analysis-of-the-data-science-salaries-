from scipy.stats import skew, kurtosis  # Importing skew and kurtosis functions from scipy.stats

def calculate_summary_statistics(df, chosen_variable):
    """
    Calculate summary statistics including sum, mean, standard deviation, skewness, and kurtosis
    for a chosen variable in a DataFrame.

    Parameters:
    - df (DataFrame): The pandas DataFrame containing the data.
    - chosen_variable (str): The name of the variable for which to calculate statistics.

    Returns:
    - summary (Series): Summary statistics including count, mean, std, min, 25%, 50%, 75%, and max.
    - skewness (float): Skewness of the variable.
    - kurt (float): Kurtosis of the variable.
    """
    # Summary statistics using describe()
    summary = df[chosen_variable].describe()

    # Calculate skewness and kurtosis
    skewness = skew(df[chosen_variable])
    kurt = kurtosis(df[chosen_variable])

    return summary, skewness, kurt

# Chosen variable
chosen_variable = 'salary_in_usd'

# Calculate summary statistics
summary, skewness, kurt = calculate_summary_statistics(df, chosen_variable)

# Print summary statistics
print("Summary Statistics:")
print(summary)
print(f"\nSkewness: {skewness}")
print(f"Kurtosis: {kurt}")

from sklearn.base import BaseEstimator, TransformerMixin
from sklearn.preprocessing import OneHotEncoder
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

class FeatureEncoder(BaseEstimator, TransformerMixin):
    """
    Custom transformer for one-hot encoding categorical features 
    computing correlation 
    between the salary and encoded features.
    Methods used :
    fit(self, X, y=None): Fit the transformer.
    transform(self, X): Transform the input DataFrame by one-hot encoding categorical features
                        and computing correlation with salary.
    """

    def fit(self, X, y=None):
        """
        Fit the transformer.

        Parameters:
        - X (DataFrame): Input DataFrame containing categorical features.

        Returns:
        - self: Returns an instance of the transformer.
        """
        return self 
    
    def transform(self,X):
        """
        Transform the input DataFrame by one-hot encoding categorical features
        and computing correlation with salary.

        Parameters:
        - X (DataFrame): Input DataFrame containing categorical features.

        Returns:
        - X (DataFrame): Transformed DataFrame with encoded features.
        """
        encoder=OneHotEncoder()
        
        # Encodes 'experience_level' column using one-hot encoding
        matrix = encoder.fit_transform(X[['experience_level']]).toarray()
        experience_level_column_names= ["SE", "MI", "EN", "EX"]

        # Creates columns for 'experience_level'
        for i in range(len(matrix.T)): 
            X[experience_level_column_names[i]]=matrix.T[i]

        # Encodes 'employment_type' column using one-hot encoding
        matrix= encoder.fit_transform(X[['employment_type']]).toarray()
        employment_type_column_names=["FT", "CT", "FL", "PT"]

        # Creates columns for 'employment_type'
        for i in range(len(matrix.T)): 
            X[employment_type_column_names[i]]=matrix.T[i]
            
        # Encodes 'company_size' column using one-hot encoding
        matrix= encoder.fit_transform(X[['company_size']]).toarray()
        company_size_column_names=["L", "S", "M"]

        # Creates columns for 'company_size'
        for i in range(len(matrix.T)): 
            X[company_size_column_names[i]]=matrix.T[i]
        
        # Compute correlation between salary and encoded features
        correlation = X[['salary_in_usd'] + experience_level_column_names].corr()
        
        # Plot heatmap
        plt.figure(figsize=(10, 8))
        sns.heatmap(correlation, annot=True, cmap='coolwarm', fmt=".2f")
        plt.title("Correlation between Salary and Encoded Features")
        plt.show()
        
        return X
